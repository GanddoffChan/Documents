\[\begin{aligned}
    p&\to q\\
    \iff\neg p&\lor q\\
    \iff q&\lor\neg p\\
    \iff\neg\neg q&\lor\neg p\\
    \iff\neg(\neg q)&\lor\neg p\\
    \iff\neg q&\to\neg p\qed\\
\end{aligned}\]
