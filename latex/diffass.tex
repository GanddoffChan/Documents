\[x=\tan\theta\implies\frac{\dd x}{\dd\theta}=\sec^2\theta=1+\tan^2\theta=1+x^2\]
\[\frac{\dd y}{\dd\theta}
=\frac{\dd y}{\dd x}\frac{\dd x}{\dd\theta}
=\frac{\dd y}{\dd x}\left(1+x^2\right)
\implies\left(1+x^2\right)\frac{\dd y}{\dd x}=\frac{\dd y}{\dd\theta}\qed\]

\[\begin{aligned}
	\left(1+x^2\right)\frac{\dd y}{\dd x}&=\frac{\dd y}{\dd\theta}\\
	\left(1+x^2\right)\frac{\dd^2y}{\dd x^2}+2x\frac{\dd y}{\dd x}
	&=\frac{\dd^2y}{\dd\theta^2}\frac{\dd\theta}{\dd x}\\
	\left(1+x^2\right)\frac{\dd^2y}{\dd x^2}
	&=\frac{\dd^2y}{\dd\theta^2}\left(1+x^2\right)^{-1}-2x\frac{\dd y}{\dd x}\\
	\left(1+x^2\right)^3\frac{\dd^2y}{\dd x^2}
	&=\frac{\dd^2y}{\dd\theta^2}\left(1+x^2\right)
	-2x\frac{\dd y}{\dd x}\left(1+x^2\right)^2\\
	\left(1+x^2\right)^3\frac{\dd^2y}{\dd x^2}
	&=\left(1+x^2\right)\frac{\dd^2y}{\dd\theta^2}
	-2x\left(1+x^2\right)\frac{\dd y}{\dd\theta}\\
\end{aligned}\]

\[\begin{aligned}
	\left(1+x^2\right)^3\frac{\dd^2y}{\dd x^2}
	+2\left(1+x^2\right)^2(1+x)\frac{\dd y}{\dd x}
	-3\left(1+x^2\right)y
	&=x^2+6x-1\\
	\left(1+x^2\right)\frac{\dd^2y}{\dd\theta^2}
	-2x\left(1+x^2\right)\frac{\dd y}{\dd\theta}
	+2\left(1+x^2\right)(1+x)\frac{\dd y}{\dd\theta}
	-3\left(1+x^2\right)y
	&=x^2+6x-1\\
	\left(1+x^2\right)\frac{\dd^2y}{\dd\theta^2}
	+2\left(1+x^2\right)\frac{\dd y}{\dd\theta}
	-3\left(1+x^2\right)y
	&=x^2+6x-1\\
	\frac{\dd^2y}{\dd\theta^2}
	+2\frac{\dd y}{\dd\theta}
	-3y
	&=\frac{x^2+6x-1}{1+x^2}\\
\end{aligned}\]

\[\begin{aligned}
	\frac{x^2+6x-1}{1+x^2}&=\frac{\tan^2\theta+6\tan\theta-1}{\sec^2\theta}\\
	&=\cos^2\theta\left(\frac{\sin^2\theta}{\cos^2\theta}
	+6\frac{\sin\theta}{\cos\theta}-1\right)\\
	&=\sin^2\theta+6\sin\theta\cos\theta-\cos^2\theta\\
	&=3(2\sin\theta\cos\theta)-\left(\cos^2\theta-\sin^2\theta\right)\\
	&=3\sin2\theta-\cos2\theta
\end{aligned}\]

\[\therefore\frac{\dd^2y}{\dd\theta^2}+2\frac{\dd y}{\dd\theta}-3y
=3\sin2\theta-\cos2\theta\qed\]

\[\begin{aligned}
	y_p&=p\sin2\theta+q\cos2\theta\\
	\frac{\dd y_p}{\dd\theta}&=2p\cos2\theta-2q\sin2\theta\\
	\frac{\dd^2y_p}{\dd\theta^2}&=-4p\sin2\theta-4q\cos2\theta\\
\end{aligned}\]

\[\begin{aligned}
	\frac{\dd^2y_p}{\dd\theta^2}+2\frac{\dd y_p}{\dd\theta}-3y_p
	&=3\sin2\theta-\cos2\theta\\
	-4p\sin2\theta-4q\cos2\theta
	+4p\cos2\theta-4q\sin2\theta
	-3p\sin2\theta-3q\cos2\theta
	&=3\sin2\theta-\cos2\theta\\
	-(4p+7q)\sin2\theta+(4p-7q)\cos2\theta
	&=3\sin2\theta-\cos2\theta\\
\end{aligned}\]

\[\begin{cases}
	4p+7q=-3\\
	4p-7q=-1
\end{cases}
\implies p=-\frac12,\,q=-\frac17\]
