\[y\propto x^3\iff y=kx^3\]
\\
\\
\[\begin{aligned}
	y_1&=k(x_1)^3\\
	3&=k(x_1)^3\\
	(x_1)^3&=\frac3k\\
	x_1&=\sqrt[3]{\frac3k}
\end{aligned}\]
\\
\\
\[\begin{aligned}
	y_2&=k(x_2)^3\\
	&=k\left(\frac{x_1}2\right)^3\\
	&=k\left(\frac{\sqrt[3]{\frac3k}}2\right)^3\\
	&=k\left(\frac{\frac3k}8\right)\\
	&=\boxed{\frac38}
\end{aligned}\]
